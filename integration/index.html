// MRaccount ID bo'yicha foydalanuvchini olish
async function getMRaccountUser(id) {
  const url = `https://firestore.googleapis.com/v1/projects/
mrplatform-9cdc0/databases/(default)/documents/users/${id}`;

  const res = await fetch(url);
  if (!res.ok) throw new Error('Foydalanuvchi topilmadi');

  const data = await res.json();
  if (!data.fields) throw new Error('Foydalanuvchi topilmadi');

  return {
    id,
    name:     data.fields?.name?.stringValue     || '',
    username: data.fields?.username?.stringValue || '',
  };
}

async function mraLogin() {
  const id    = document.getElementById('mraIdInput').value.trim();
  const errEl = document.getElementById('mraError');
  errEl.textContent = '';

  if (!id) { errEl.textContent = 'ID kiriting'; return; }

  try {
    const user = await getMRaccountUser(id);

    // ✅ Mana shu yerda foydalanuvchi bilan ishlaysiz:
    console.log(user.name);       // "Jasur Karimov"
    console.log(user.username);   // "jasur_uz"
    console.log(user.id);         // "abc123..."

    document.getElementById('mraModal').style.display = 'none';
  } catch (e) {
    errEl.textContent = e.message;
  }
}
